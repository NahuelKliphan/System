create table categorias(
	id serial,
	nombre varchar(30) unique,
	descripcion varchar(50),
	primary key (id)
);

create table productos(
	id serial,
	codigo varchar(30),
	nombre varchar(30),
	precio_costo decimal,
	precio_venta decimal,
	cantidad integer,
	descripcion varchar(50),
	foto text,
	id_categoria integer,
	primary key (id),
	foreign key (id_categoria) references categorias on delete set null
);

create table ventas(
	id serial,
	cliente_nombre varchar(30),
	fecha date,
	total decimal,
	ganancia decimal,
	primary key (id)
);

create table items(
	id serial,
    id_venta integer,
    total decimal,
    codigo varchar(30),
    nombre varchar(50),
    cantidad integer,
    precio_costo decimal,
	precio_venta decimal,
	ganancia decimal,
	primary key (id),
	foreign key (id_venta) references ventas on delete cascade
);

create table registro_precios(
	id serial primary key,
	fecha date,
	operacion varchar(20),
	tipo_valor varchar(20),
	tipo_precio varchar(20),
	valor numeric,
	anulada char(1) default 'N',
	id_categoria integer
);

create table entidades(
	id serial primary key,
	nombre varchar(50),
	direccion varchar(50),
	telefono varchar(50),
	mail varchar(50),
	cuit varchar(50),
	facebook varchar(50),
	instagram varchar(50),
	twitter varchar(50),
	facebook_link varchar(50),
	instagram_link varchar(50),
	twitter_link varchar(50),
	logo text
);
